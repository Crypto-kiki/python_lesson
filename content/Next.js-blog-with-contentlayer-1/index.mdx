## Tailwind CSS

tailwind.config.js 파일을 보면 기본 설정값이 있습니다.

```javascript
// tailwind.config.js

/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    "./src/pages/**/*.{js,ts,jsx,tsx,mdx}",
    "./src/components/**/*.{js,ts,jsx,tsx,mdx}",
    "./src/app/**/*.{js,ts,jsx,tsx,mdx}",
  ],
  theme: {
    extend: {
      backgroundImage: {
        "gradient-radial": "radial-gradient(var(--tw-gradient-stops))",
        "gradient-conic":
          "conic-gradient(from 180deg at 50% 50%, var(--tw-gradient-stops))",
      },
    },
  },
  plugins: [],
};
```

backgroundImage는 사용하지 않기 때문에, 삭제합니다.
그리고 프로젝트에 사용 할 색상을 아래와 같이 정의해줍니다.

## layout.js

### setup fonts

```javascript
// src/app/layout.js

import { Inter, Manrope } from "next/font/google";
import "./globals.css";

const inter = Inter({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-in",
});
const manrope = Manrope({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-mr",
});

export const metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body className={inter.className}>{children}</body>
    </html>
  );
}
```

1.  **`javascript
import { Inter, Manrope } from "next/font/google";`** - next/font/google 모듈에서 Inter와 Manrope라는 두 폰트 불러오기 함수를 가져옵니다. - 이 모듈은 Next.js 12.2 버전부터 도입된 폰트 최적화 기능을 사용해 Google 폰트를 쉽게 프로젝트에 추가할 수 있게 해 줍니다
2.  **`javascript
const inter = Inter({ subsets: ["latin"], display: "swap", variable: "--font-in" });`** - Inter 폰트를 설정합니다. - 여기서 subsets 옵션으로 "latin" 문자 집합을 선택하고, display 옵션으로 "swap"을 사용하여 폰트가 로드될 때까지 텍스트가 보이는 방식을 설정합니다. - variable 옵션은 CSS 변수명을 지정하며, 이 변수는 나중에 CSS에서 해당 폰트를 사용할 때 참조됩니다.
3.  **`const manrope = Manrope({ subsets: ["latin"], display: "swap", variable: "--font-mr" });`**
    - Manrope 폰트를 설정합니다.
    - 설정 옵션은 Inter 설정과 유사하지만 CSS 변수명으로는 "--font-mr"을 사용하여 Manrope 폰트를 참조할 수 있게 합니다.

그리고 `<body>`태그에 Inter와 Manrope 폰트를 적용해줍니다.

```javascript
// src/app/layout.js

import { Inter, Manrope } from "next/font/google";
import "./globals.css";

const inter = Inter({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-in",
});
const manrope = Manrope({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-mr",
});

export const metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body className={`${inter.variable} ${manrope.variable}`}>
        {children}
      </body>
    </html>
  );
}
```

### font-family

inter와 manrope 폰트를 className으로 사용하기 위해서 tailwindcss에서 fontfamily를 추가합니다.

```javascript
// tailwind.config.js

/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    "./src/pages/**/*.{js,ts,jsx,tsx,mdx}",
    "./src/components/**/*.{js,ts,jsx,tsx,mdx}",
    "./src/app/**/*.{js,ts,jsx,tsx,mdx}",
  ],
  theme: {
    extend: {
      colors: {
        dark: "#1b1b1b",
        light: "#fff",
        accent: "#7b00d3",
        accentDark: "#ffdb4d",
        gray: "#747474",
      },
      fontFamily: {
        mr: ["var(--font-mr)"],
        in: ["var(--font-in)"],
      },
    },
  },
  plugins: [],
};
```

이제 Inter, Manrope 폰트를 사용하려면 className에 'font-in' 또는 'font-mr' 속성을 사용해서 폰트를 적용 할 수 있습니다.
background-color는 light, manrope 폰트가 기본으로 적용되도록 body태그에 적용해줍니다.

```javascript
// src/app/layout.js
import { Inter, Manrope } from "next/font/google";
import "./globals.css";

const inter = Inter({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-in",
});
const manrope = Manrope({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-mr",
});

export const metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body
        className={`${inter.variable} ${manrope.variable} font-mr bg-light dark:bg-dark`}
      >
        {children}
      </body>
    </html>
  );
}
```

## Utils

src/utils/index.js를 생성합니다.

```javascript
// src/utils/index.js

export const cx = (...className) => className.filter(Boolean).join(" ");
```

cx 함수는 여러 개의 클래스 이름을 받아서 하나의 문자열로 결합해주는 역할을 합니다. 주로 웹 개발에서 HTML 요소의 클래스를 동적으로 관리할 때 사용됩니다.
즉, 이 함수는 주어진 클래스 이름 중에서 "유효한" 것들만을 골라서 공백으로 구분된 하나의 문자열로 만들어 줍니다. 여기서 "유효한" 클래스 이름이란, false, null, undefined, 빈 문자열 "", 0, NaN과 같이 JavaScript에서 거짓(falsy)으로 평가되는 값이 아닌 모든 값들을 의미합니다.

### edit layout.js

```javascript
// src/app/layout.js

import { Inter, Manrope } from "next/font/google";
import "./globals.css";
import { cx } from "@/utils";

const inter = Inter({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-in",
});
const manrope = Manrope({
  subsets: ["latin"],
  display: "swap",
  variable: "--font-mr",
});

export const metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body
        className={cx(
          inter.variable,
          manrope.variable,
          "font-mr bg-light dark:bg-dark"
        )}
      >
        {children}
      </body>
    </html>
  );
}
```

## Install Vscode extension

1.  **Tailwind CSS IntelliSense**
    - Visual Studio Code(VSCode) 환경에서 React, Redux 및 React Native 개발을 위한 코드 스니펫을 제공하는 확장 프로그램입니다. 이 확장 기능은 개발자가 React 관련 코드를 더 빠르고 효율적으로 작성할 수 있게 도와줍니다.
    - JavaScript의 ES7+ 문법을 포함하여, React 컴포넌트, React Hooks, Redux 함수 등을 쉽게 삽입할 수 있는 다양한 스니펫을 포함하고 있습니다.
    - 예를 들어, rcc를 입력하고 자동완성 기능을 사용하면 클래스 컴포넌트의 기본 구조가 생성됩니다.
2.  **ES7+ React/Redux/React-Native snippets**
    - VSCode에서 Tailwind CSS를 사용할 때 더 나은 자동완성, 색상 미리보기, 클래스 이름에 대한 도움말 등을 제공하는 확장 프로그램입니다.
    - 개발자가 CSS 클래스를 입력할 때, 가능한 옵션을 자동으로 제시하고, 해당 클래스가 적용될 스타일을 미리보기로 보여줍니다.
    - 프로젝트의 tailwind.config.js 파일 설정을 기반으로 한 맞춤 제안도 제공합니다.
